#### Pre-request Script（自动获取Token）

**Collection级别Pre-request Script：**

```javascript
// 自动登录获取Token（如果token为空或即将过期）
const token = pm.environment.get("token");
const username = pm.environment.get("test_username");
const password = pm.environment.get("test_password");

if (!token || token === "") {
    const loginRequest = {
        url: pm.environment.get("base_url") + "/api/mer/auth/login",
        method: "POST",
        header: {
            "Content-Type": "application/json"
        },
        body: {
            mode: "raw",
            raw: JSON.stringify({
                username: username,
                password: password
            })
        }
    };

    pm.sendRequest(loginRequest, function (err, res) {
        if (res && res.code === 200) {
            pm.environment.set("token", res.json().data.token);
            console.log("Token已更新");
        }
    });
}
```
