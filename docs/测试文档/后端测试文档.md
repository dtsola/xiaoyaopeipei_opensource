# tests/test_crud/test_sku.py

import pytest
from sqlalchemy.orm import Session
from app.crud.crud_sku import crud_sku
from app.schemas.sku import SkuCreate
from app.models.sku import Sku

class TestSkuCRUD:
    """SKU CRUD测试"""

    def test_create_sku_success(self, db: Session):
        """测试创建SKU成功"""
        sku_data = SkuCreate(
            name="测试配置",
            device_type="laptop",
            price=5999.00,
            cpu="Intel i5-12500H",
            gpu="RTX 3050",
            ram="16GB DDR4",
            storage="512GB SSD",
        )
        result = crud_sku.create(db, obj_in=sku_data, merchant_id=1)

        assert result.name == "测试配置"
        assert result.device_type == "laptop"
        assert result.price == 5999.00
        assert result.id is not None

    def test_get_sku_by_id(self, db: Session):
        """测试根据ID获取SKU"""
        sku = crud_sku.get_by_id(db, id=1)
        assert sku is not None
        assert sku.id == 1

    def test_get_sku_by_merchant_with_pagination(self, db: Session):
        """测试分页获取商家的SKU列表"""
        items, total = crud_sku.get_by_merchant(
            db=db,
            merchant_id=1,
            skip=0,
            limit=10
        )
        assert isinstance(items, list)
        assert isinstance(total, int)
        assert len(items) <= 10

    def test_update_sku(self, db: Session):
        """测试更新SKU"""
        sku = crud_sku.get_by_id(db, id=1)
        updated_sku = crud_sku.update(
            db,
            db_obj=sku,
            obj_in={"price": 6999.00}
        )
        assert updated_sku.price == 6999.00

    def test_delete_sku_soft(self, db: Session):
        """测试软删除SKU"""
        sku = crud_sku.get_by_id(db, id=1)
        deleted_sku = crud_sku.delete(db, db_obj=sku)
        assert deleted_sku.status == "inactive"

    def test_batch_delete_skus(self, db: Session):
        """测试批量删除SKU"""
        success_count, failed_count = crud_sku.batch_delete(
            db=db,
            merchant_id=1,
            sku_ids=[1, 2, 3]
        )
        assert success_count >= 0
        assert failed_count >= 0
