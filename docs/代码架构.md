### **app/core/logger.py（Loguru配置）⭐**

```python
"""
Loguru日志配置
"""
import sys
from pathlib import Path
from loguru import logger
from app.core.config import settings

# 日志目录
LOG_DIR = Path("logs")
LOG_DIR.mkdir(exist_ok=True)


def setup_logger():
    """
    配置Loguru日志
    """
    # 移除默认的handler
    logger.remove()

    # 控制台输出（开发环境）
    if settings.APP_DEBUG:
        logger.add(
            sys.stdout,
            level=settings.LOG_LEVEL,
            format=settings.LOG_FORMAT,
            colorize=True,
        )

    # 应用日志文件（INFO及以上）
    logger.add(
        LOG_DIR / "app_{time:YYYY-MM-DD}.log",
        level="INFO",
        format=settings.LOG_FORMAT,
        rotation=settings.LOG_ROTATION,  # 每天00:00轮转
        retention=settings.LOG_RETENTION,  # 保留30天
        compression="zip",  # 压缩旧日志
        encoding="utf-8",
        enqueue=True,  # 异步写入
    )

    # 错误日志文件（ERROR及以上）
    logger.add(
        LOG_DIR / "error_{time:YYYY-MM-DD}.log",
        level="ERROR",
        format=settings.LOG_FORMAT,
        rotation=settings.LOG_ROTATION,
        retention=settings.LOG_RETENTION,
        compression="zip",
        encoding="utf-8",
        enqueue=True,
        backtrace=True,  # 显示堆栈跟踪
        diagnose=True,  # 显示变量值
    )

    # 访问日志文件（单独记录API访问）
    logger.add(
        LOG_DIR / "access_{time:YYYY-MM-DD}.log",
        level="INFO",
        format="{time:YYYY-MM-DD HH:mm:ss} | {message}",
        rotation=settings.LOG_ROTATION,
        retention=settings.LOG_RETENTION,
        compression="zip",
        encoding="utf-8",
        enqueue=True,
        filter=lambda record: "access" in record["extra"],
    )

    logger.info("✅ Loguru日志系统初始化成功")


def get_logger(name: str = __name__):
    """
    获取日志实例
    
    Args:
        name: 日志名称（通常使用 __name__）
    
    Returns:
        logger实例
    """
    return logger.bind(name=name)
```

```python
"""
日志中间件（Loguru）
"""
import time
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware
from app.core.logger import get_logger

logger = get_logger(__name__)


class LoggingMiddleware(BaseHTTPMiddleware):
    """
    请求日志中间件
    """

    async def dispatch(self, request: Request, call_next):
        # 记录请求开始时间
        start_time = time.time()

        # 获取请求信息
        method = request.method
        url = request.url.path
        client_ip = request.client.host if request.client else "unknown"

        # 处理请求
        try:
            response = await call_next(request)
            status_code = response.status_code

            # 计算处理时间
            process_time = (time.time() - start_time) * 1000

            # 记录访问日志
            logger.bind(access=True).info(
                f"{client_ip} - {method} {url} - {status_code} - {process_time:.2f}ms"
            )

            # 添加响应头
            response.headers["X-Process-Time"] = f"{process_time:.2f}ms"

            return response

        except Exception as e:
            # 计算处理时间
            process_time = (time.time() - start_time) * 1000

            # 记录错误日志
            logger.error(
                f"❌ 请求处理失败: {method} {url} - {client_ip} - {process_time:.2f}ms\n"
                f"错误: {str(e)}"
            )
            raise
```

```python
        return Tongyi(
            dashscope_api_key=settings.QWEN_API_KEY,
            model_name=settings.QWEN_MODEL,
            temperature=settings.QWEN_TEMPERATURE,
            max_tokens=settings.QWEN_MAX_TOKENS,
        )

    @staticmethod
    def create_llm(provider: str = "qwen"):
        """
        创建LLM实例（支持多模型切换）

        Args:
            provider: 模型提供商（qwen/doubao/etc）
        """
        if provider == "qwen":
            return LLMFactory.create_qwen_llm()
        # 后续可扩展其他模型
        else:
            raise ValueError(f"不支持的LLM提供商: {provider}")
```

### **app/services/ai/chains/extraction_chain.py（信息提取链）**

```python
"""
信息提取链
"""
from langchain.chains import LLMChain
from langchain.output_parsers import PydanticOutputParser
from pydantic import BaseModel, Field

from app.services.ai.base import llm
from app.services.ai.prompts.extraction_prompt import EXTRACTION_PROMPT
from app.core.logger import get_logger

logger = get_logger(__name__)


# 提取信息的数据模型
class ExtractedInfo(BaseModel):
    """AI提取的信息"""
    budget: str | None = Field(None, description="预算")
    device_type: str | None = Field(None, description="设备类型")
    usage: str | None = Field(None, description="用途")
    requirements: str | None = Field(None, description="具体需求")
    brand: str | None = Field(None, description="品牌偏好")
    portable: str | None = Field(None, description="便携性需求")
    confidence: float = Field(..., description="置信度")


class ExtractionChain:
    """信息提取链"""

    def __init__(self):
        """初始化信息提取链"""
        self.parser = PydanticOutputParser(pydantic_object=ExtractedInfo)
        self.chain = LLMChain(
            llm=llm,
            prompt=EXTRACTION_PROMPT,
            verbose=False,  # 生产环境设为False
        )
        logger.info("✅ 信息提取链初始化成功")

    async def extract(self, user_message: str, conversation_history: str = "") -> ExtractedInfo:
        """
        提取用户信息

        Args:
            user_message: 用户消息
            conversation_history: 对话历史

        Returns:
            ExtractedInfo: 提取的信息
        """
        logger.debug(f"🤖 开始提取信息: {user_message}")

        try:
            # 调用链
            result = await self.chain.ainvoke(
                {
                    "user_message": user_message,
                    "conversation_history": conversation_history,
                }
            )

            # 解析输出
            extracted_text = result["text"]
            logger.debug(f"📤 AI原始输出: {extracted_text}")

            # 使用解析器解析JSON
            info = self.parser.parse(extracted_text)

            logger.info(f"✅ 信息提取成功: confidence={info.confidence}")
            return info

        except Exception as e:
            logger.error(f"❌ 信息提取失败: {str(e)}")
            # 返回低置信度默认值
            return ExtractedInfo(confidence=0.0)


# 全局实例
extraction_chain = ExtractionChain()
```

```python
"""
对话记忆管理
"""
from langchain.memory import ConversationBufferMemory
from langchain.memory.chat_message import HumanMessage, AIMessage
from app.core.logger import get_logger

logger = get_logger(__name__)


class ConversationMemoryManager:
    """对话记忆管理器"""

    def __init__(self, session_id: str):
        """
        初始化对话记忆

        Args:
            session_id: 会话ID
        """
        self.session_id = session_id
        self.memory = ConversationBufferMemory(
            memory_key="conversation_history",
            return_messages=True,
        )
        logger.debug(f"💾 创建对话记忆: {session_id}")

    def add_user_message(self, message: str):
        """添加用户消息"""
        self.memory.chat_memory.add_message(HumanMessage(content=message))
        logger.debug(f"📝 添加用户消息: {message}")

    def add_ai_message(self, message: str):
        """添加AI消息"""
        self.memory.chat_memory.add_message(AIMessage(content=message))
        logger.debug(f"🤖 添加AI消息: {message}")

    def get_history(self) -> str:
        """获取对话历史（用于Prompt）"""
        history = self.memory.load_memory_variables({})
        messages = history.get("conversation_history", [])

        # 转换为字符串格式
        history_text = "\n".join(
            f"{msg.type}: {msg.content}" for msg in messages[-5:]  # 只保留最近5条
        )
        return history_text

    def clear(self):
        """清空对话记忆"""
        self.memory.clear()
        logger.debug(f"🗑️ 清空对话记忆: {self.session_id}")
```

### **app/services/ai_service.py（AI服务封装）**

```python
"""
AI服务（封装LangChain）
"""
from app.services.ai.chains.extraction_chain import extraction_chain
from app.services.ai.memory.conversation_memory import ConversationMemoryManager
from app.core.logger import get_logger
from app.core.cache import get_cache

logger = get_logger(__name__)


class AIService:
    """AI服务"""

    def __init__(self):
        """初始化AI服务"""
        self.extraction_chain = extraction_chain
        self.cache = get_cache()

    async def extract_user_intent(
        self,
        session_id: str,
        user_message: str,
    ) -> dict:
        """
        提取用户意图和信息

        Args:
            session_id: 会话ID
            user_message: 用户消息

        Returns:
            dict: 提取的信息
        """
        logger.info(f"🤖 提取用户意图: {session_id}")

        # 获取对话记忆
        memory = self._get_memory(session_id)
        conversation_history = memory.get_history()

        # 调用信息提取链
        try:
            info = await self.extraction_chain.extract(
                user_message=user_message,
                conversation_history=conversation_history,
            )

            # 保存到对话记忆
            memory.add_user_message(user_message)

            # 转换为字典
            result = {
                "budget": info.budget,
                "device_type": info.device_type,
                "usage": info.usage,
                "requirements": info.requirements,
                "confidence": info.confidence,
            }

            logger.info(f"✅ 意图提取成功: {result}")
            return result

        except Exception as e:
            logger.error(f"❌ 意图提取失败: {str(e)}")
            raise

    def _get_memory(self, session_id: str) -> ConversationMemoryManager:
        """
        获取对话记忆（带缓存）

        Args:
            session_id: 会话ID

        Returns:
            ConversationMemoryManager
        """
        cache_key = f"memory:{session_id}"

        # 尝试从缓存获取
        cached = self.cache.get(cache_key)
        if cached:
            return cached

        # 创建新的对话记忆
        memory = ConversationMemoryManager(session_id)
        self.cache.set(cache_key, memory, ttl=1800)  # 30分钟

        return memory


# 全局实例
ai_service = AIService()
```
